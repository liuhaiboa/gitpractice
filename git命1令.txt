git init 将当前目录初始化为本地仓库

git clone url 将远程仓库拷贝到当前目录  过程中建立远程连接

git add . 将当前目录下所有文件添加到工作区（包括文件夹）

git commit -m‘提交信息（必填）'

git push 【远程仓库地址】【远程分支名称】：【本地分支名称】

git pull 【远程仓库地址】【远程分支名称】：【本地分支名称】

git remote -v 查看远程仓库

git branch 【分支名称】 新建分支

git branch 查看本地分支

git branch -r 查看远程分支

git checkout 【分支名称】切换（检出）分支

git  merge 【A分支名称】 将A分支合并到当前分支
